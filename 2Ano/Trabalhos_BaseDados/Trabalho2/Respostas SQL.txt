CREATE TABLE Editoras (
    NIF VARCHAR(20) PRIMARY KEY,
    Nome VARCHAR(100),
    Email VARCHAR(100)
);

-- Criação da tabela Genero
CREATE TABLE Genero (
    Nome_genero VARCHAR(50) PRIMARY KEY
);

-- Criação da tabela Livros
CREATE TABLE Livros (
    ISBN VARCHAR(20) PRIMARY KEY,
    Titulo VARCHAR(100),
    Nome_genero VARCHAR(50),
    FOREIGN KEY (Nome_genero) REFERENCES Genero(Nome_genero)
);

-- Criação da tabela Autores
CREATE TABLE Autores (
    Id_autor SERIAL PRIMARY KEY,
    Nome VARCHAR(100),
    Nacionalidade VARCHAR(50)
);


-- Criação da tabela Encomenda
CREATE TABLE Encomenda (
    Valor DECIMAL(10, 2),
    Dia INT,
    Mes INT,
    Ano INT,
    NIF_Editora VARCHAR(20),
    PRIMARY KEY (Dia, Mes, Ano, NIF_Editora),
    FOREIGN KEY (NIF_Editora) REFERENCES Editoras(NIF)
);


-- Criação da tabela Exemplares de Livros
CREATE TABLE Exemplares_de_Livros (
    Id_exemplar SERIAL PRIMARY KEY,
    ISBN VARCHAR(20),
    Link VARCHAR(200),
    FOREIGN KEY (ISBN) REFERENCES Livros(ISBN)
);

-- Criação da tabela Leitor
CREATE TABLE Leitor (
    NIF VARCHAR(20) PRIMARY KEY,
    Nome VARCHAR(100),
    Email VARCHAR(100),
    Nacionalidade VARCHAR(50),
    Tipo VARCHAR(20) CHECK (Tipo IN ('Regular', 'Frequente'))
);

-- Criação da tabela Amizade
CREATE TABLE Amizade (
    NIF_Amigo VARCHAR(20),
    NIF_Leitor VARCHAR(20),
    PRIMARY KEY (NIF_Amigo, NIF_Leitor),
    FOREIGN KEY (NIF_Amigo) REFERENCES Leitor(NIF),
    FOREIGN KEY (NIF_Leitor) REFERENCES Leitor(NIF)
);

-- Criação da tabela Avaliação
CREATE TABLE Avaliacao (
    Estrelas INT CHECK (Estrelas BETWEEN 1 AND 5),
    Comentario TEXT,
    ISBN VARCHAR(20),
    NIF_Leitor VARCHAR(20),
    PRIMARY KEY (ISBN, NIF_Leitor),
    FOREIGN KEY (ISBN) REFERENCES Livros(ISBN),
    FOREIGN KEY (NIF_Leitor) REFERENCES Leitor(NIF)
);

-- Criação da tabela Faturação
CREATE TABLE Faturacao (
    Mes INT,
    Ano INT,
    Valor DECIMAL(10, 2),
    NIF_Leitor VARCHAR(20),
    PRIMARY KEY (Mes, Ano, NIF_Leitor),
    FOREIGN KEY (NIF_Leitor) REFERENCES Leitor(NIF)
);

-- Criação da tabela Aluguer
CREATE TABLE Aluguer (
    Id_exemplar INT,
    NIF_Leitor VARCHAR(20),
    Valor DECIMAL(10, 2),
    NºSemana INT,
    Ano INT,
    PRIMARY KEY (Id_exemplar, NºSemana, Ano),
    FOREIGN KEY (Id_exemplar) REFERENCES Exemplares_de_Livros(Id_exemplar),
    FOREIGN KEY (NIF_Leitor) REFERENCES Leitor(NIF)
);

--criação da tabela livro_Autor
CREATE TABLE Livro_Autor (
    ISBN VARCHAR(20),
    Id_autor INT,
    PRIMARY KEY (ISBN, Id_autor),
    FOREIGN KEY (ISBN) REFERENCES Livros(ISBN),
    FOREIGN KEY (Id_autor) REFERENCES Autores(Id_autor)
);
